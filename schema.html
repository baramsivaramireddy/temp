CREATE TABLE IF NOT EXISTS users ( user_id TEXT PRIMARY KEY, name TEXT NOT NULL, email TEXT UNIQUE NOT NULL, address TEXT, phone TEXT, password TEXT NOT NULL, role TEXT CHECK(role IN ('customer', 'admin')) NOT NULL ); CREATE TABLE IF NOT EXISTS menu ( item_id INTEGER PRIMARY KEY AUTOINCREMENT, name TEXT NOT NULL, category TEXT NOT NULL, price REAL NOT NULL, availability INTEGER DEFAULT 1 ); CREATE TABLE IF NOT EXISTS orders ( order_id INTEGER PRIMARY KEY AUTOINCREMENT, user_id TEXT NOT NULL, items TEXT NOT NULL, total REAL NOT NULL, status TEXT DEFAULT 'Pending', FOREIGN KEY(user_id) REFERENCES users(user_id) );
