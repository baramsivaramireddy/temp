from menu import view_menu, search_menu
from order import place_order, view_order_history

def customer_dashboard(user):
    while True:
        print("\n--- Customer Dashboard ---")
        print("1. View Menu")
        print("2. Search Menu")
        print("3. Place Order")
        print("4. Order History")
        print("5. Logout")

        choice = input("Enter choice: ")

        if choice == "1":
            view_menu()
        elif choice == "2":
            keyword = input("Enter food name/category: ")
            search_menu(keyword)
        elif choice == "3":
            place_order(user["user_id"])
        elif choice == "4":
            view_order_history(user["user_id"])
        elif choice == "5":
            print("Logging out...")
            break
        else:
            print("Invalid choice!")
